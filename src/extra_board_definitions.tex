% Sizes
\pgfmathsetmacro{\fieldwidth}{5}
\pgfmathsetmacro{\fieldheight}{2}

\pgfmathsetmacro{\marketboardwidth}{6}
\pgfmathsetmacro{\marketboardheight}{\fieldheight*7}

\newcommand{\marketboard}[1]{
	\draw[#1,ultra thick] (0,0) rectangle (\marketboardwidth,\marketboardheight);
	\foreach \x in {1,...,6}{
		\draw[#1] (0.5,0.5*\fieldheight+\x*\fieldheight-\fieldheight) rectangle (0.5+\fieldwidth,0.5\fieldheight+\x*\fieldheight);
		\node [text width=\fieldwidth cm] at (0.5+0.5*\fieldwidth,\x*\fieldheight) {
			\begin{center}
				{\Large \bf \x}
			\end{center}
		};
	};
}
\newcommand{\startspace}[2]{
	\node [text width=\fieldwidth cm] at (0.7+0.5*\fieldwidth,#2*\fieldheight-0.2) {
		\includegraphics[width=0.5 cm]{#1}
	};	
}
\newcommand{\secondarystartspace}[2]{
	\node [text width=\fieldwidth cm] at (1.3+0.5*\fieldwidth,#2*\fieldheight-0.2) {
		\includegraphics[width=0.5 cm]{#1}
	};	
}
