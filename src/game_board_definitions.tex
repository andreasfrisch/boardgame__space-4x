\newcommand{\drawhexat}[2]{
	\draw[black, fill=white]
		(#1-0.5*\hexwidth,#2) --
		(#1-0.25*\hexwidth,#2+\triangleheight) --
		(#1+0.25*\hexwidth,#2+\triangleheight) --
		(#1+0.5*\hexwidth,#2) --
		(#1+0.25*\hexwidth,#2-\triangleheight) --
		(#1-0.25*\hexwidth,#2-\triangleheight) --
		cycle;
}

\def\shapeHex{
	(-0.5*\hexwidth,0) --
	(-0.25*\hexwidth,\triangleheight) --
	(0.25*\hexwidth,\triangleheight) --
	(0.5*\hexwidth,0) --
	(0.25*\hexwidth,-\triangleheight) --
	(-0.25*\hexwidth,-\triangleheight) --
	cycle
}
